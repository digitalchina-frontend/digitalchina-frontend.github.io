# 主题配置

## 配置 Logo

```js{1,11-13}
import { defaultTheme, defineUserConfig } from 'vuepress'

export default defineUserConfig({
  locales: {
    '/': {
      lang: 'zh-CN',
      title: '现代前端开发指南',
      description: '大道至简，让前端开发更简单',
    },
  },
  theme: defaultTheme({
    logo: 'images/logo.png',
  }),
})
```
